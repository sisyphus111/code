Qt 01：Qt 基础与常用控件

Qt 是一个跨平台的 C++ GUI 框架。机考中可能会要求用 Qt 实现简单的图形界面程序，掌握基本控件、信号槽机制和布局即可应对。

核心概念：Qt 程序由 QApplication、主窗口、控件、布局和信号槽组成。控件通过信号槽机制通信，布局管理器自动调整控件位置。

1. Qt 程序的基本结构

一个最简单的 Qt 程序包含：

#include <QApplication>
#include <QWidget>

int main(int argc, char* argv[]) {
    QApplication app(argc, argv);  // 创建应用程序对象

    QWidget window;                // 创建主窗口
    window.setWindowTitle("Hello Qt");
    window.resize(400, 300);
    window.show();                 // 显示窗口

    return app.exec();             // 进入事件循环
}
  • QApplication：管理整个应用程序，处理事件循环
  • QWidget：所有 UI 组件的基类
  • app.exec()：启动事件循环，程序保持运行直到所有窗口关闭

3. 信号与槽（Signals & Slots）

信号槽是 Qt 的核心机制，用于对象之间的通信。当某个事件发生时，对象发出信号，槽函数被调用响应。

3.1 基本用法

#include <QApplication>
#include <QPushButton>
#include <QMessageBox>

int main(int argc, char* argv[]) {
    QApplication app(argc, argv);

    QPushButton button("点击我");

    // 连接信号和槽：按钮点击时弹出消息框
    QObject::connect(&button, &QPushButton::clicked, []() {
        QMessageBox::information(nullptr, "提示", "按钮被点击了！");
    });

    button.show();
    return app.exec();
}

QObject::connect() 的参数：

  • 发送者对象指针
  • 信号（用 &ClassName::signalName）
  • 接收者对象指针或 lambda 函数
  • 槽函数（可选，如果第三个参数是 lambda 则不需要）

3.2 常用信号

控件信号含义

QLineEdittextChanged(QString)文本改变
QLineEditreturnPressed()按下回车键
QCheckBoxstateChanged(int)选中状态改变
QComboBoxcurrentIndexChanged(int)选项改变
QSpinBoxvalueChanged(int)数值改变

4. 布局管理器（Layouts）

布局管理器自动管理控件的位置和大小，无需手动设置坐标。

4.1 三种基本布局

布局类效果用途
QVBoxLayout垂直排列从上到下排列控件
QHBoxLayout水平排列从左到右排列控件
QGridLayout网格布局按行列排列控件

4.2 使用布局示例

#include <QApplication>
#include <QWidget>
#include <QVBoxLayout>
#include <QHBoxLayout>
#include <QLabel>
#include <QLineEdit>
#include <QPushButton>

int main(int argc, char* argv[]) {
    QApplication app(argc, argv);

    QWidget window;
    window.setWindowTitle("登录界面");

    // 创建控件
    QLabel* userLabel = new QLabel("用户名：");
    QLineEdit* userEdit = new QLineEdit();

    QLabel* passLabel = new QLabel("密码：");
    QLineEdit* passEdit = new QLineEdit();
    passEdit->setEchoMode(QLineEdit::Password);  // 密码模式

    QPushButton* loginBtn = new QPushButton("登录");
    QPushButton* cancelBtn = new QPushButton("取消");

    // 创建布局
    QVBoxLayout* mainLayout = new QVBoxLayout();

    // 用户名行
    QHBoxLayout* userLayout = new QHBoxLayout();
    userLayout->addWidget(userLabel);
    userLayout->addWidget(userEdit);

    // 密码行
    QHBoxLayout* passLayout = new QHBoxLayout();
    passLayout->addWidget(passLabel);
    passLayout->addWidget(passEdit);

    // 按钮行
    QHBoxLayout* btnLayout = new QHBoxLayout();
    btnLayout->addWidget(loginBtn);
    btnLayout->addWidget(cancelBtn);

    // 组合布局
    mainLayout->addLayout(userLayout);
    mainLayout->addLayout(passLayout);
    mainLayout->addLayout(btnLayout);

    window.setLayout(mainLayout);
    window.show();

    return app.exec();
}

5. 完整示例：简单计算器

#include <QApplication>
#include <QWidget>
#include <QVBoxLayout>
#include <QHBoxLayout>
#include <QLineEdit>
#include <QPushButton>
#include <QLabel>

int main(int argc, char* argv[]) {
    QApplication app(argc, argv);

    QWidget window;
    window.setWindowTitle("简单计算器");

    // 创建控件
    QLineEdit* num1Edit = new QLineEdit();
    QLineEdit* num2Edit = new QLineEdit();
    QLabel* resultLabel = new QLabel("结果：");

    QPushButton* addBtn = new QPushButton("+");
    QPushButton* subBtn = new QPushButton("-");
    QPushButton* mulBtn = new QPushButton("×");
    QPushButton* divBtn = new QPushButton("÷");

    // 连接信号槽
    QObject::connect(addBtn, &QPushButton::clicked, [=]() {
        double a = num1Edit->text().toDouble();
        double b = num2Edit->text().toDouble();
        resultLabel->setText("结果：" + QString::number(a + b));
    });

    QObject::connect(subBtn, &QPushButton::clicked, [=]() {
        double a = num1Edit->text().toDouble();
        double b = num2Edit->text().toDouble();
        resultLabel->setText("结果：" + QString::number(a - b));
    });

    QObject::connect(mulBtn, &QPushButton::clicked, [=]() {
        double a = num1Edit->text().toDouble();
        double b = num2Edit->text().toDouble();
        resultLabel->setText("结果：" + QString::number(a * b));
    });

    QObject::connect(divBtn, &QPushButton::clicked, [=]() {
        double a = num1Edit->text().toDouble();
        double b = num2Edit->text().toDouble();
        if (b != 0) {
            resultLabel->setText("结果：" + QString::number(a / b));
        } else {
            resultLabel->setText("结果：错误（除数为0）");
        }
    });

    // 布局
    QVBoxLayout* mainLayout = new QVBoxLayout();
    mainLayout->addWidget(new QLabel("数字1："));
    mainLayout->addWidget(num1Edit);
    mainLayout->addWidget(new QLabel("数字2："));
    mainLayout->addWidget(num2Edit);

    QHBoxLayout* btnLayout = new QHBoxLayout();
    btnLayout->addWidget(addBtn);
    btnLayout->addWidget(subBtn);
    btnLayout->addWidget(mulBtn);
    btnLayout->addWidget(divBtn);

    mainLayout->addLayout(btnLayout);
    mainLayout->addWidget(resultLabel);

    window.setLayout(mainLayout);
    window.resize(300, 200);
    window.show();

    return app.exec();
}

6. 编译 Qt 程序

6.1 使用 qmake（推荐）

创建 .pro 项目文件：

# calculator.pro
QT += core gui widgets
TARGET = calculator
TEMPLATE = app
SOURCES += main.cpp

编译步骤：

qmake calculator.pro
make
./calculator

6.2 使用 CMake

cmake_minimum_required(VERSION 3.16)
project(calculator)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_AUTOMOC ON)

find_package(Qt6 REQUIRED COMPONENTS Core Gui Widgets)

add_executable(calculator main.cpp)
target_link_libraries(calculator Qt6::Core Qt6::Gui Qt6::Widgets)

编译：

mkdir build && cd build
cmake ..
make
./calculator

8. Qt 机考检查清单

  • 窗口标题和大小是否设置？
  • 控件是否正确添加到布局？
  • 信号槽是否正确连接？
  • lambda 捕获是否正确（通常用 [=] 值捕获所有变量）？
  • 是否调用了 window.show() 和 app.exec()？
  • 编译时是否包含必要的 Qt 模块（QT += widgets）？
掌握标准：能用 Qt 快速搭建包含输入框、按钮、列表的界面；能正确连接信号槽；能编译运行 Qt 程序。
