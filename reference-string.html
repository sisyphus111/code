附录 D：字符串操作详解

C++ 字符串的常用操作、查找、统计和子串处理的完整参考。

1. 字符串基本操作

1.1 创建和初始化

#include <string>
using namespace std;

string s1 = "hello";           // 字面量初始化
string s2("world");            // 构造函数
string s3(5, 'a');             // "aaaaa"（5 个 'a'）
string s4(s1);                 // 复制构造
string s5 = s1 + " " + s2;     // 拼接：hello world

1.2 长度和访问

string s = "hello";

// 长度
size_t len = s.length();       // 5
size_t sz = s.size();          // 5（和 length() 一样）
bool empty = s.empty();        // false

// 访问单个字符
char first = s[0];             // 'h'（不检查边界）
char last = s[s.length() - 1]; // 'o'
char c = s.at(2);              // 'l'（检查边界，越界抛异常）

// 修改字符
s[0] = 'H';                    // s = "Hello"

2. 查找字符和字符串

2.1 查找单个字符

string s = "hello world";

// find：从前往后找第一个
size_t pos = s.find('o');           // 4（第一个 'o' 的位置）
size_t pos2 = s.find('o', 5);       // 7（从位置 5 开始找）
size_t not_found = s.find('x');     // string::npos（没找到）

// rfind：从后往前找第一个
size_t last = s.rfind('o');         // 7（最后一个 'o' 的位置）

// 判断是否存在
if (s.find('o') != string::npos) {
    cout << "找到了 'o'\n";
}

2.2 查找子串

string s = "hello world";

// 查找子串
size_t pos = s.find("world");       // 6
size_t pos2 = s.find("World");      // string::npos（区分大小写）

// 从指定位置开始查找
size_t pos3 = s.find("o", 5);       // 7（从位置 5 开始）

// 反向查找子串
size_t last = s.rfind("o");         // 7

2.3 查找任意字符（find_first_of）

string s = "hello world";

// 查找第一个元音字母
size_t pos = s.find_first_of("aeiou");  // 1（'e' 的位置）

// 查找第一个数字
string text = "abc123def";
size_t digit_pos = text.find_first_of("0123456789");  // 3

// 查找第一个不是元音的字符
size_t consonant = s.find_first_not_of("aeiou");  // 0（'h'）

2.4 find 函数族总结

函数作用示例
find(c)查找字符或子串s.find('o')
find(s, pos)从 pos 开始查找s.find('o', 5)
rfind(c)反向查找s.rfind('o')
find_first_of(chars)查找 chars 中任意字符s.find_first_of("aeiou")
find_last_of(chars)反向查找 chars 中任意字符s.find_last_of("aeiou")
find_first_not_of(chars)查找第一个不在 chars 中的字符s.find_first_not_of(" ")
find_last_not_of(chars)反向查找第一个不在 chars 中的字符s.find_last_not_of(" ")
重要：所有 find 函数找不到时返回 string::npos（通常是 -1 的无符号表示），必须检查返回值！

3. 统计字符个数

3.1 统计单个字符

#include <algorithm>
#include <string>
using namespace std;

string s = "hello world";

// 方法 1：使用 count
int cnt = count(s.begin(), s.end(), 'o');  // 2

// 方法 2：手动遍历
int cnt2 = 0;
for (char c : s) {
    if (c == 'o') cnt2++;
}

// 方法 3：查找所有出现位置
int cnt3 = 0;
size_t pos = 0;
while ((pos = s.find('o', pos)) != string::npos) {
    cnt3++;
    pos++;  // 移到下一个位置
}

3.2 统计多种字符

string s = "hello world";

// 统计元音字母个数
int vowel_count = 0;
for (char c : s) {
    if (c == 'a' || c == 'e' || c == 'i' || c == 'o' || c == 'u') {
        vowel_count++;
    }
}

// 或者用 set 判断
set<char> vowels = {'a', 'e', 'i', 'o', 'u'};
int cnt = 0;
for (char c : s) {
    if (vowels.count(c)) cnt++;
}

3.3 统计每个字符的出现次数

string s = "hello world";

// 方法 1：数组（只适合 ASCII）
int freq[256] = {};
for (char c : s) {
    freq[(unsigned char)c]++;
}
cout << "字母 'o' 出现了 " << freq['o'] << " 次\n";

// 方法 2：map（适合所有字符）
map<char, int> freq_map;
for (char c : s) {
    freq_map[c]++;
}
cout << "字母 'o' 出现了 " << freq_map['o'] << " 次\n";

4. 子串操作

4.1 substr 详解

string s = "hello world";

// substr(pos) - 从 pos 到末尾
string sub1 = s.substr(6);      // "world"

// substr(pos, len) - 从 pos 开始取 len 个字符
string sub2 = s.substr(0, 5);   // "hello"
string sub3 = s.substr(6, 3);   // "wor"

// 注意：pos 是下标（从 0 开始），不是迭代器！
// 正确：
string sub4 = s.substr(1);      // "ello world"

// 错误示范：
// string sub5 = s.substr(s.begin() + 1);  // 编译错误！substr 要的是整数
常见错误：substr 的参数是整数下标，不是迭代器！substr(1) 表示从下标 1 开始，而不是 s.begin() + 1。

4.2 substr 参数说明

调用方式含义示例（s = "hello"）结果
substr()整个字符串s.substr()"hello"
substr(pos)从 pos 到末尾s.substr(2)"llo"
substr(pos, len)从 pos 开始取 len 个s.substr(1, 3)"ell"

4.3 常见子串操作

string s = "hello world";

// 取前 n 个字符
string first_n = s.substr(0, 5);        // "hello"

// 取后 n 个字符
string last_n = s.substr(s.length() - 5);  // "world"

// 去掉前 n 个字符
string without_first = s.substr(6);     // "world"

// 去掉后 n 个字符
string without_last = s.substr(0, s.length() - 6);  // "hello"

// 取中间部分
string middle = s.substr(3, 5);         // "lo wo"

5. 字符串修改

5.1 插入和删除

string s = "hello world";

// 插入
s.insert(5, ",");              // "hello, world"
s.insert(0, "say ");           // "say hello, world"

// 删除
s.erase(0, 4);                 // 删除前 4 个字符："hello, world"
s.erase(5);                    // 删除位置 5 及之后的所有："hello"

// 替换
s.replace(0, 5, "goodbye");    // 将 [0, 5) 替换为 "goodbye"

5.2 追加和拼接

string s = "hello";

// 追加
s += " world";                 // "hello world"
s.append("!");                 // "hello world!"
s.push_back('?');              // "hello world!?"（追加单个字符）

// 拼接
string s1 = "hello";
string s2 = "world";
string s3 = s1 + " " + s2;     // "hello world"

5.3 清空和重置

string s = "hello";

s.clear();                     // s = ""（清空，长度为 0）
s = "";                        // 同上

// 替换整个字符串
s.assign("world");             // s = "world"
s.assign(5, 'a');              // s = "aaaaa"

6. 字符串比较

6.1 比较操作符

string s1 = "apple";
string s2 = "banana";

bool eq = (s1 == s2);          // false
bool ne = (s1 != s2);          // true
bool lt = (s1 < s2);           // true（字典序）
bool gt = (s1 > s2);           // false

6.2 compare 函数

string s1 = "apple";
string s2 = "banana";

int result = s1.compare(s2);
// result < 0: s1 < s2
// result == 0: s1 == s2
// result > 0: s1 > s2

// 比较子串
int cmp = s1.compare(0, 3, s2, 0, 3);  // 比较 "app" 和 "ban"

7. 字符串转换

7.1 大小写转换

#include <algorithm>
#include <cctype>

string s = "Hello World";

// 转大写
transform(s.begin(), s.end(), s.begin(), ::toupper);  // "HELLO WORLD"

// 转小写
transform(s.begin(), s.end(), s.begin(), ::tolower);  // "hello world"

// 手动转换
for (char& c : s) {
    c = toupper(c);  // 或 tolower(c)
}

7.2 去除空白

string s = "  hello world  ";

// 去除前导空白
size_t start = s.find_first_not_of(" \t\n\r");
if (start != string::npos) {
    s = s.substr(start);
}

// 去除尾部空白
size_t end = s.find_last_not_of(" \t\n\r");
if (end != string::npos) {
    s = s.substr(0, end + 1);
}

// 完整的 trim 函数
string trim(const string& str) {
    size_t first = str.find_first_not_of(" \t\n\r");
    if (first == string::npos) return "";
    size_t last = str.find_last_not_of(" \t\n\r");
    return str.substr(first, last - first + 1);
}

9. 字符串分割

9.1 按空白分割

#include <sstream>
#include <vector>

string s = "hello world test";
vector<string> words;

istringstream iss(s);
string word;
while (iss >> word) {
    words.push_back(word);
}
// words = {"hello", "world", "test"}

9.2 按特定字符分割

string s = "apple,banana,cherry";
vector<string> parts;

istringstream iss(s);
string part;
while (getline(iss, part, ',')) {
    parts.push_back(part);
}
// parts = {"apple", "banana", "cherry"}

9.3 通用分割函数

vector<string> split(const string& s, char delimiter) {
    vector<string> tokens;
    istringstream stream(s);
    string token;
    while (getline(stream, token, delimiter)) {
        tokens.push_back(token);
    }
    return tokens;
}

// 使用
auto parts = split("a,b,c,d", ',');
