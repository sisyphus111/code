不同类型的 INF 与 NEG_INF

这里的 INF 和 NEG_INF 分别表示该类型能够保存的最大值和最小值，而不是数学意义上的正负无穷。它们适合赋值、初始化和比较，不应参与加减乘除等算术运算。

#include <limits>

constexpr short SHORT_INF =
    std::numeric_limits<short>::max();
constexpr short SHORT_NEG_INF =
    std::numeric_limits<short>::lowest();

constexpr int INT_INF =
    std::numeric_limits<int>::max();
constexpr int INT_NEG_INF =
    std::numeric_limits<int>::lowest();

constexpr long LONG_INF =
    std::numeric_limits<long>::max();
constexpr long LONG_NEG_INF =
    std::numeric_limits<long>::lowest();

constexpr long long LL_INF =
    std::numeric_limits<long long>::max();
constexpr long long LL_NEG_INF =
    std::numeric_limits<long long>::lowest();

constexpr float FLOAT_INF =
    std::numeric_limits<float>::max();
constexpr float FLOAT_NEG_INF =
    std::numeric_limits<float>::lowest();

constexpr double DOUBLE_INF =
    std::numeric_limits<double>::max();
constexpr double DOUBLE_NEG_INF =
    std::numeric_limits<double>::lowest();
注意： 整数的最小值通常比最大值的相反数小 1，所以 NEG_INF 必须使用 lowest()，不能写成 -INF。 对浮点类型也要使用 lowest()；它们的 min() 表示最小正数。
类型INFNEG_INF说明
shortmax()lowest()通常为 16 位
intmax()lowest()通常为 32 位
longmax()lowest()可能是 32 位或 64 位
long longmax()lowest()至少为 64 位
floatmax()lowest()最大和最小有限值
doublemax()lowest()最大和最小有限值
比较例外： 上述边界值可以正常使用 <、>、<=、>= 和 ==。 但浮点数如果是 NaN，这些比较不会遵循普通大小关系。

场景二：逐行读取命令并处理

示例支持 add 数字、sum、print、clear 和 exit。新增命令时，直接增加一个分支。

#include <bits/stdc++.h>
using namespace std;

int main() {
    vector<int> values;
    string line;

    while (getline(cin, line)) {
        stringstream ss(line);
        string command;
        ss >> command;

        if (command.empty()) {
            continue;
        }

        if (command == "add") {
            int value;
            if (!(ss >> value)) {
                cout << "Error: add needs an integer\n";
                continue;
            }
            values.push_back(value);

            long long sum = 0;
            for (int value : values) {
                sum += value;
            }
            cout << sum << '\n';

                if (i > 0) cout << ' ';
                cout << values[i];
            }
            cout << '\n';
        } else if (command == "clear") {
            values.clear();
        } else if (command == "exit") {
            break;
        } else {
            cout << "Error: unknown command\n";
        }
    }

    return 0;
}

场景四：命令行选项、文件读写与交互命令

同一套程序既可以在终端中显示提示符并接收命令，也可以从文件批量读取命令、把结果写入另一个文件。

选项作用
-i FILE / --input FILE从文件读取命令；不提供时从键盘读取
-o FILE / --output FILE把命令结果写入文件；不提供时输出到终端
--prompt TEXT修改交互提示符，默认是 $
--no-prompt不显示交互提示符，适合管道输入
-h / --help显示帮助并退出

运行示例：

./program
./program --prompt "command> "
./program --no-prompt
./program --input commands.txt
./program -i commands.txt -o result.txt

命令仍然使用上一场景的处理方式：add 数字 添加数据，sum 求和，print 输出全部数据，clear 清空，exit 退出。

#include <bits/stdc++.h>
using namespace std;

int main(int argc, char* argv[]) {
    string input_path;
    string output_path;
    string prompt = "$ ";
    bool show_prompt = true;

    for (int i = 1; i < argc; ++i) {
        string option = argv[i];

        if (option == "-h" || option == "--help") {
            cout << "Usage: program [options]\n"
                 << "  -i, --input FILE    read commands from FILE\n"
                 << "  -o, --output FILE   write results to FILE\n"
                 << "      --prompt TEXT   set interactive prompt\n"
                 << "      --no-prompt     hide interactive prompt\n";
            return 0;
        } else if (option == "-i" || option == "--input") {
            if (i + 1 == argc) {
                cerr << "Error: " << option << " needs a file\n";
                return 1;
            }
            input_path = argv[++i];
        } else if (option == "-o" || option == "--output") {
            if (i + 1 == argc) {
                cerr << "Error: " << option << " needs a file\n";
                return 1;
            }
            output_path = argv[++i];
        } else if (option == "--prompt") {
            if (i + 1 == argc) {
                cerr << "Error: --prompt needs text\n";
                return 1;
            }
            prompt = argv[++i];
        } else if (option == "--no-prompt") {
            show_prompt = false;
        } else {
            cerr << "Error: unknown option " << option << '\n';
            return 1;
        }
    }

    ifstream fin;
    if (!input_path.empty()) {
        fin.open(input_path);
        if (!fin) {
            cerr << "Error: cannot open " << input_path << '\n';
            return 1;
        }
        show_prompt = false;
    }

    ofstream fout;
    if (!output_path.empty()) {
        fout.open(output_path);
        if (!fout) {
            cerr << "Error: cannot open " << output_path << '\n';
            return 1;
        }
    }

    istream& input = input_path.empty() ? cin : fin;
    ostream& output = output_path.empty() ? cout : fout;

    vector<int> values;
    string line;

    while (true) {
        if (show_prompt) {
            cerr << prompt << flush;
        }
        if (!getline(input, line)) {
            break;
        }

        stringstream ss(line);
        string command;
        ss >> command;

        if (command.empty()) {
            continue;
        }

        if (command == "add") {
            int value;
            if (!(ss >> value)) {
                cerr << "Error: add needs an integer\n";
                continue;
            }
            values.push_back(value);
        } else if (command == "sum") {
            long long sum = 0;
            for (int value : values) {
                sum += value;
            }
            output << sum << '\n';
        } else if (command == "print") {
            for (int i = 0; i < (int)values.size(); ++i) {
                if (i > 0) output << ' ';
                output << values[i];
            }
            output << '\n';
        } else if (command == "clear") {
            values.clear();
        } else if (command == "exit") {
            break;
        } else {
            cerr << "Error: unknown command\n";
        }
    }

    return 0;
}
为什么提示符写到 cerr： cerr << "$ " << flush 会在程序等待输入前立刻显示提示符， 同时不会把 $ 混进标准输出或结果文件。真正的命令结果统一写到 output。

查找第一个 >= x

int lower_bound_manual(const vector<int>& a, int x) {
    int l = 0, r = a.size();
    while (l < r) {
        int m = l + (r - l) / 2;
        if (a[m] >= x) r = m;
        else l = m + 1;
    }
    return l;
}

long long binary_min(long long l, long long r) {
    while (l < r) {
        long long m = l + (r - l) / 2;
        if (check(m)) r = m;
        else l = m + 1;
    }
    return l;
}

二分答案 - 最大可行

long long binary_max(long long l, long long r) {
    while (l < r) {
        long long m = l + (r - l + 1) / 2;
        if (check(m)) l = m;
        else r = m - 1;
    }
    return l;
}

DFS、BFS 与回溯

DFS 网格

int n, m;
vector<vector<int>> vis;
int dx[4] = {1, -1, 0, 0};
int dy[4] = {0, 0, 1, -1};

bool inside(int x, int y) {
    return x >= 0 && x < n && y >= 0 && y < m;
}

void dfs(int x, int y) {
    vis[x][y] = 1;
    for (int d = 0; d < 4; ++d) {
        int nx = x + dx[d], ny = y + dy[d];
        if (!inside(nx, ny) || vis[nx][ny]) continue;
        dfs(nx, ny);
    }
}

BFS 网格最短路

vector<vector<int>> dist(n, vector<int>(m, -1));
queue<pii> q;

dist[sx][sy] = 0;
q.push({sx, sy});

while (!q.empty()) {
    auto [x, y] = q.front();
    q.pop();
    
    for (int d = 0; d < 4; ++d) {
        int nx = x + dx[d], ny = y + dy[d];
        if (!inside(nx, ny) || dist[nx][ny] != -1) continue;
        dist[nx][ny] = dist[x][y] + 1;
        q.push({nx, ny});
    }
}

回溯：组合

vector<int> path;
vector<vector<int>> ans;

void backtrack(int start, int n, int k) {
    if ((int)path.size() == k) {
        ans.push_back(path);
        return;
    }
    for (int i = start; i <= n; ++i) {
        path.push_back(i);
        backtrack(i + 1, n, k);
        path.pop_back();
    }
}

回溯：排列

vector<int> path, used;
vector<vector<int>> ans;

void permute(vector<int>& nums) {
    if (path.size() == nums.size()) {
        ans.push_back(path);
        return;
    }
    for (int i = 0; i < (int)nums.size(); ++i) {
        if (used[i]) continue;
        used[i] = 1;
        path.push_back(nums[i]);
        permute(nums);
        path.pop_back();
        used[i] = 0;
    }
}

并查集与拓扑排序

并查集

struct DSU {
    vector<int> p, sz;
    
    DSU(int n) : p(n), sz(n, 1) {
        iota(p.begin(), p.end(), 0);
    }
    
    int find(int x) {
        return p[x] == x ? x : p[x] = find(p[x]);
    }
    
    bool unite(int a, int b) {
        a = find(a); b = find(b);
        if (a == b) return false;
        if (sz[a] < sz[b]) swap(a, b);
        p[b] = a;
        sz[a] += sz[b];
        return true;
    }
    
    bool same(int a, int b) {
        return find(a) == find(b);
    }
};

拓扑排序

vector<int> topo_sort(int n, vector<vector<int>>& g) {
    vector<int> indeg(n, 0);
    for (int u = 0; u < n; ++u) {
        for (int v : g[u]) indeg[v]++;
    }
    
    queue<int> q;
    for (int i = 0; i < n; ++i) {
        if (indeg[i] == 0) q.push(i);
    }
    
    vector<int> order;
    while (!q.empty()) {
        int u = q.front(); q.pop();
        order.push_back(u);
        for (int v : g[u]) {
            if (--indeg[v] == 0) q.push(v);
        }
    }
    
    return order; // size < n 说明有环
}

最短路与最小生成树

Dijkstra 单源最短路

用于边权非负的图；邻接表元素为 {终点, 边权}。

vector<ll> dijkstra(int n, int s, vector<vector<pii>>& g) {
    vector<ll> dist(n, LL_INF);
    priority_queue<pair<ll, int>, vector<pair<ll, int>>, greater<>> pq;
    
    dist[s] = 0;
    pq.push({0, s});
    
    while (!pq.empty()) {
        auto [d, u] = pq.top(); pq.pop();
        if (d != dist[u]) continue;
        
        for (auto [v, w] : g[u]) {
            if (dist[v] > d + w) {
                dist[v] = d + w;
                pq.push({dist[v], v});
            }
        }
    }
    
    return dist;
}

Prim 最小生成树

用于连通无向图；成功时返回总权重与生成树的边，图不连通时返回 nullopt。

optional<pair<ll, vector<pair<int, int>>>> prim(
    int n, const vector<vector<pii>>& g
) {
    if (n == 0) {
        return make_pair(0LL, vector<pair<int, int>>{});
    }

    // {边权, 当前点, 父节点}
    using State = tuple<ll, int, int>;
    priority_queue<State, vector<State>, greater<State>> pq;
    vector<bool> used(n, false);
    vector<pair<int, int>> tree_edges;
    ll total = 0;
    int count = 0;

    pq.push({0, 0, -1});
    while (!pq.empty()) {
        auto [weight, u, parent] = pq.top();
        pq.pop();
        if (used[u]) continue;

        used[u] = true;
        total += weight;
        ++count;
        if (parent != -1) tree_edges.push_back({parent, u});

        for (auto [v, edge_weight] : g[u]) {
            if (!used[v]) pq.push({edge_weight, v, u});
        }
    }

    if (count != n) return nullopt;
    return make_pair(total, tree_edges);
}

Kruskal 最小生成树

复用上面的 DSU；适合输入本身就是边列表的情况，图不连通时返回 nullopt。

struct Edge {
    int u, v;
    ll weight;
};

optional<pair<ll, vector<Edge>>> kruskal(
    int n, vector<Edge> edges
) {
    if (n == 0) return make_pair(0LL, vector<Edge>{});

    sort(edges.begin(), edges.end(), [](const Edge& a, const Edge& b) {
        return a.weight < b.weight;
    });

    DSU dsu(n);
    ll total = 0;
    vector<Edge> tree_edges;

    for (const Edge& edge : edges) {
        if (!dsu.unite(edge.u, edge.v)) continue;
        total += edge.weight;
        tree_edges.push_back(edge);
        if ((int)tree_edges.size() == n - 1) break;
    }

    if ((int)tree_edges.size() != n - 1) return nullopt;
    return make_pair(total, tree_edges);
}

树状数组

struct Fenwick {
    int n;
    vector<ll> bit;
    
    Fenwick(int n) : n(n), bit(n + 1, 0) {}
    
    void add(int idx, ll val) {
        for (++idx; idx <= n; idx += idx & -idx) 
            bit[idx] += val;
    }
    
    ll sum(int idx) {
        ll res = 0;
        for (++idx; idx > 0; idx -= idx & -idx) 
            res += bit[idx];
        return res;
    }
    
    ll range_sum(int l, int r) {
        return sum(r) - (l == 0 ? 0 : sum(l - 1));
    }
};

线段树

struct SegTree {
    int n;
    vector<ll> tr;
    
    SegTree(const vector<int>& a) : n(a.size()), tr(n * 4, 0) {
        build(1, 0, n - 1, a);
    }
    
    void build(int o, int l, int r, const vector<int>& a) {
        if (l == r) { tr[o] = a[l]; return; }
        int m = (l + r) / 2;
        build(o * 2, l, m, a);
        build(o * 2 + 1, m + 1, r, a);
        tr[o] = tr[o * 2] + tr[o * 2 + 1];
    }
    
    void update(int o, int l, int r, int idx, int val) {
        if (l == r) { tr[o] = val; return; }
        int m = (l + r) / 2;
        if (idx <= m) update(o * 2, l, m, idx, val);
        else update(o * 2 + 1, m + 1, r, idx, val);
        tr[o] = tr[o * 2] + tr[o * 2 + 1];
    }
    
    ll query(int o, int l, int r, int ql, int qr) {
        if (ql <= l && r <= qr) return tr[o];
        int m = (l + r) / 2;
        ll res = 0;
        if (ql <= m) res += query(o * 2, l, m, ql, qr);
        if (qr > m) res += query(o * 2 + 1, m + 1, r, ql, qr);
        return res;
    }
};

ll mod_pow(ll a, ll e, ll mod) {
    ll r = 1 % mod;
    a %= mod;
    while (e > 0) {
        if (e & 1) r = r * a % mod;
        a = a * a % mod;
        e >>= 1;
    }
    return r;
}

x & -x             // 保留最低位的 1
x | (1 << k)       // 第 k 位置 1
x & ~(1 << k)      // 第 k 位置 0
x ^ (1 << k)       // 第 k 位翻转
(x >> k) & 1       // 取第 k 位

// 统计 1 的个数
__builtin_popcount(x)      // int
__builtin_popcountll(x)    // long long

// 枚举子集
for (int s = mask; s; s = (s - 1) & mask) {
    // s 是 mask 的子集
}
