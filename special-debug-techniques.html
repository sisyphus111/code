打印调试技巧

基础调试宏

#include <bits/stdc++.h>

    #define debug(x) cerr << #x << " = " << (x) << endl

增强版调试宏

#ifdef LOCAL
    #define debug(x) cerr << #x << " = " << (x) << endl
    #define debugArr(arr, n) cerr << #arr << " = "; for(int i=0;i<(n);++i) cerr << (arr)[i] << " "; cerr << endl
    #define debugVec(v) cerr << #v << " = "; for(auto x : v) cerr << x << " "; cerr << endl
    #define debugMap(m) cerr << #m << " = {"; for(auto [k,v] : m) cerr << k << ":" << v << " "; cerr << "}" << endl
    #define debugPair(p) cerr << #p << " = (" << (p).first << ", " << (p).second << ")" << endl
#else
    #define debug(x)
    #define debugArr(arr, n)
    #define debugVec(v)
    #define debugMap(m)
    #define debugPair(p)
#endif

int main() {
    vector<int> v = {1, 2, 3, 4, 5};
    debugVec(v);  // 输出: v = 1 2 3 4 5
    
    map<string, int> mp = {{"a", 1}, {"b", 2}};
    debugMap(mp);  // 输出: mp = {a:1 b:2}
    
    pair<int, int> p = {3, 4};
    debugPair(p);  // 输出: p = (3, 4)
    
    return 0;
}

提交时：g++ -std=c++17 solution.cpp（调试语句自动消失）

输出调用栈信息

int depth = 0;

void dfs(int u, vector<vector<int>>& g, vector<bool>& vis) {
    depth++;
    cerr << string(depth * 2, ' ') << "-> dfs(" << u << ")\n";
    
    vis[u] = true;
    for (int v : g[u]) {
        if (!vis[v]) {
            dfs(v, g, vis);
        }
    }
    
    cerr << string(depth * 2, ' ') << "<- dfs(" << u << ")\n";
    depth--;
}

// 输出示例：
//   -> dfs(0)
//     -> dfs(1)
//       -> dfs(2)
//       <- dfs(2)
//     <- dfs(1)
//   <- dfs(0)

二分调试法

什么是二分调试？

当代码很长，不知道哪里出错时，用二分法定位：注释掉一半代码，看问题是否还在。

步骤

  1. 确认输入没问题（打印输入）
  2. 在代码中间位置打印中间结果
  3. 如果中间结果正确，问题在后半部分
  4. 如果中间结果错误，问题在前半部分
  5. 重复缩小范围，直到找到出错行

实战示例

int main() {
    int n;
    cin >> n;
    debug(n);  // 步骤 1：确认输入
    
    vector<int> a(n);
    for (int i = 0; i < n; ++i) {
        cin >> a[i];
    }
    debugVec(a);  // 步骤 2：确认输入数组
    
    // 一堆复杂逻辑...
    
    int result = 0;
    for (int i = 0; i < n; ++i) {
        result += a[i] * 2;
    }
    debug(result);  // 步骤 3：检查中间结果
    
    // 更多逻辑...
    
    cout << result << "\n";
    return 0;
}

对拍脚本

什么是对拍？

对拍（Stress Test）：用暴力算法和优化算法对比，找出数据使两者答案不一致。

对拍三件套：
  • gen.cpp：随机数据生成器
  • brute.cpp：暴力算法（保证正确）
  • solution.cpp：优化算法（待验证）

数据生成器 (gen.cpp)

#include <bits/stdc++.h>
using namespace std;

int main(int argc, char* argv[]) {
    // 使用命令行参数作为种子，确保每次生成不同数据
    int seed = argc > 1 ? atoi(argv[1]) : time(0);
    srand(seed);
    
    int n = rand() % 10 + 1;  // [1, 10]
    cout << n << "\n";
    
    for (int i = 0; i < n; ++i) {
        int x = rand() % 100 + 1;  // [1, 100]
        cout << x << " ";
    }
    cout << "\n";
    
    return 0;
}

暴力算法 (brute.cpp)

#include <bits/stdc++.h>
using namespace std;

// O(n^2) 暴力，但保证正确
int main() {
    int n;
    cin >> n;
    vector<int> a(n);
    for (int i = 0; i < n; ++i) {
        cin >> a[i];
    }
    
    int answer = 0;
    for (int i = 0; i < n; ++i) {
        for (int j = i + 1; j < n; ++j) {
            if (a[i] + a[j] == 10) {
                answer++;
            }
        }
    }
    
    cout << answer << "\n";
    return 0;
}

对拍脚本 (Linux/Mac - duipai.sh)

#!/bin/bash

# 编译所有程序
g++ -std=c++17 -O2 gen.cpp -o gen
g++ -std=c++17 -O2 brute.cpp -o brute
g++ -std=c++17 -O2 solution.cpp -o solution

# 对拍循环
for ((i=1; i<=1000; i++)); do
    echo "Test case $i"
    
    # 生成数据
    ./gen $i > input.txt
    
    # 运行两个程序
    ./brute < input.txt > brute_output.txt
    ./solution < input.txt > solution_output.txt
    
    # 比较输出
    if diff brute_output.txt solution_output.txt > /dev/null; then
        echo "Test $i: AC"
    else
        echo "Test $i: WA"
        echo "Input:"
        cat input.txt
        echo "Brute output:"
        cat brute_output.txt
        echo "Solution output:"
        cat solution_output.txt
        break
    fi
done

echo "All tests passed!"

对拍脚本 (Windows - duipai.bat)

@echo off
g++ -std=c++17 -O2 gen.cpp -o gen.exe
g++ -std=c++17 -O2 brute.cpp -o brute.exe
g++ -std=c++17 -O2 solution.cpp -o solution.exe

for /l %%i in (1,1,1000) do (
    echo Test case %%i
    gen.exe %%i > input.txt
    brute.exe < input.txt > brute_output.txt
    solution.exe < input.txt > solution_output.txt
    
    fc brute_output.txt solution_output.txt > nul
    if errorlevel 1 (
        echo Test %%i: WA
        type input.txt
        type brute_output.txt
        type solution_output.txt
        goto :end
    )
    echo Test %%i: AC
)

echo All tests passed!
:end
pause

C++11 随机数（更好）

#include <random>

int main() {
    random_device rd;
    mt19937 gen(rd());  // Mersenne Twister 生成器
    
    // 均匀分布 [1, 100]
    uniform_int_distribution<int> dis(1, 100);
    int x = dis(gen);
    
    // 浮点数 [0.0, 1.0)
    uniform_real_distribution<double> dis2(0.0, 1.0);
    double y = dis2(gen);
    
    return 0;
}

生成随机数组

// 生成 n 个 [L, R] 的随机整数
void generateArray(int n, int L, int R) {
    cout << n << "\n";
    for (int i = 0; i < n; ++i) {
        int x = rand() % (R - L + 1) + L;
        cout << x << " ";
    }
    cout << "\n";
}

// 生成无重复的随机数组
void generateUniqueArray(int n, int L, int R) {
    set<int> used;
    cout << n << "\n";
    while (used.size() < n) {
        int x = rand() % (R - L + 1) + L;
        if (!used.count(x)) {
            used.insert(x);
            cout << x << " ";
        }
    }
    cout << "\n";
}

生成随机树

// 生成 n 个节点的随机树
void generateTree(int n) {
    cout << n << "\n";
    for (int i = 2; i <= n; ++i) {
        int parent = rand() % (i - 1) + 1;  // 父节点在 [1, i-1]
        cout << parent << " " << i << "\n";
    }
}

// 生成随机有向图
void generateDAG(int n, int m) {
    cout << n << " " << m << "\n";
    set<pair<int, int>> edges;
    while (edges.size() < m) {
        int u = rand() % n + 1;
        int v = rand() % n + 1;
        if (u != v && u < v) {  // 保证 u < v，形成 DAG
            edges.insert({u, v});
        }
    }
    for (auto [u, v] : edges) {
        cout << u << " " << v << "\n";
    }
}

生成随机字符串

// 生成长度为 n 的随机小写字母字符串
void generateString(int n) {
    for (int i = 0; i < n; ++i) {
        char c = 'a' + rand() % 26;
        cout << c;
    }
    cout << "\n";
}

// 生成随机括号序列（保证合法）
void generateBrackets(int n) {
    string s;
    int balance = 0;
    for (int i = 0; i < 2 * n; ++i) {
        if (balance == 0 || (balance < n && rand() % 2 == 0)) {
            s += '(';
            balance++;
        } else {
            s += ')';
            balance--;
        }
    }
    cout << s << "\n";
}

边界用例构造

常见边界情况

务必测试的边界：
  • 最小输入：n = 1, 空数组
  • 最大输入：n = 10^5, 10^6
  • 全部相同：所有元素都是 0 或都是最大值
  • 递增/递减：有序数组
  • 交替：1, -1, 1, -1...
  • 极值：INT_MIN, INT_MAX, LLONG_MAX

构造 Hack 数据

// 针对哈希表的 Hack（最坏情况 O(n)）
// 生成大量哈希冲突的数据
void hackUnorderedMap() {
    const int mod = 1000003;  // 常见的哈希表模数
    int n = 100000;
    cout << n << "\n";
    for (int i = 0; i < n; ++i) {
        cout << i * mod << " ";  // 都映射到同一个桶
    }
    cout << "\n";
}

// 针对快排的 Hack（最坏情况 O(n^2)）
// 生成导致快排退化的数据
void hackQuickSort() {
    int n = 100000;
    cout << n << "\n";
    for (int i = 1; i <= n; ++i) {
        cout << i << " ";  // 已排序，naive 快排会退化
    }
    cout << "\n";
}

调试工具

assert 断言

#include <cassert>

int main() {
    int n;
    cin >> n;
    assert(n >= 1 && n <= 100000);  // 检查输入范围
    
    vector<int> a(n);
    for (int i = 0; i < n; ++i) {
        cin >> a[i];
        assert(a[i] >= 0);  // 检查非负
    }
    
    // assert 在条件为 false 时会终止程序并报错
    return 0;
}

sanitizers（强烈推荐）

# Address Sanitizer：检测内存错误（数组越界、use-after-free）
g++ -std=c++17 -O2 -fsanitize=address solution.cpp
./a.out

# Undefined Behavior Sanitizer：检测未定义行为（整数溢出、除零）
g++ -std=c++17 -O2 -fsanitize=undefined solution.cpp
./a.out

# 同时使用
g++ -std=c++17 -O2 -fsanitize=address,undefined solution.cpp
./a.out
提示：sanitizers 能检测到很多隐藏的 bug，建议本地测试时总是开启。

调试检查清单

遇到 bug 时的排查顺序：
  1. ✅ 确认样例输入输出正确
  2. ✅ 检查数组越界（下标是否从 0 还是 1 开始）
  3. ✅ 检查整数溢出（是否需要 long long）
  4. ✅ 检查未初始化变量
  5. ✅ 检查多组数据是否清空
  6. ✅ 打印中间结果，二分定位
  7. ✅ 用暴力算法对拍
  8. ✅ 测试边界情况
  9. ✅ 使用 sanitizers 检测
