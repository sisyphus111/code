6. 时间复杂度参考

数据规模 n允许的复杂度典型算法
n ≤ 10O(n!), O(2ⁿ)全排列、子集枚举
n ≤ 20O(2ⁿ), O(n²×2ⁿ)状态压缩 DP
n ≤ 100O(n³)Floyd、三重循环
n ≤ 1,000O(n²)二重循环、朴素 DP
n ≤ 10,000O(n²)仍可接受
n ≤ 100,000O(n log n)排序、堆、线段树
n ≤ 1,000,000O(n), O(n log n)哈希表、单调栈、排序
n ≤ 10,000,000O(n)线性扫描、前缀和
经验：C++ 程序每秒约执行 10⁸-10⁹ 次基本操作。时限 1 秒时，O(n²) 通常在 n ≤ 10⁴ 时安全。

7. 常用数学公式

7.1 组合数与排列数

// 阶乘
long long factorial(int n) {
    long long result = 1;
    for (int i = 2; i <= n; ++i) {
        result *= i;
    }
    return result;
}

// 组合数 C(n, k) = n! / (k! * (n-k)!)
long long combination(int n, int k) {
    if (k > n - k) k = n - k;  // 优化
    long long result = 1;
    for (int i = 0; i < k; ++i) {
        result = result * (n - i) / (i + 1);
    }
    return result;
}

// 排列数 P(n, k) = n! / (n-k)!
long long permutation(int n, int k) {
    long long result = 1;
    for (int i = 0; i < k; ++i) {
        result *= (n - i);
    }
    return result;
}

7.2 最大公约数与最小公倍数

#include <numeric>  // C++17

int g = std::gcd(a, b);       // 最大公约数
int l = std::lcm(a, b);       // 最小公倍数（C++17）

// 手写版本
int gcd(int a, int b) {
    return b == 0 ? a : gcd(b, a % b);
}

long long lcm(long long a, long long b) {
    return a / gcd(a, b) * b;  // 注意先除后乘，避免溢出
}

7.3 快速幂

// 计算 a^b mod m
long long power_mod(long long a, long long b, long long m) {
    long long result = 1;
    a %= m;
    while (b > 0) {
        if (b & 1) result = result * a % m;
        a = a * a % m;
        b >>= 1;
    }
    return result;
}

11. 字符串处理

11.1 字符串分割

#include <sstream>

// 按空白分割
std::string line = "hello world test";
std::istringstream iss(line);
std::string word;
while (iss >> word) {
    std::cout << word << '\n';
}

// 按特定字符分割
std::string s = "a,b,c,d";
std::string token;
std::istringstream tokenStream(s);
while (std::getline(tokenStream, token, ',')) {
    std::cout << token << '\n';
}
